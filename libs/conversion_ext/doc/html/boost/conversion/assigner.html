Struct template assigner

boost::conversion::assigner

Synopsis

// In header: <boost/conversion/assign_to.hpp>

template<typename Target,   // target type of the conversion.
         typename Source,   // source type of the conversion.
         typename Enable = void  // A dummy template parameter that can be
                                 // used for SFINAE.
         > 
struct assigner :
  public boost::conversion::assigner_cp< Target, Source, Enable >
{
};

Description

Default customization point for assign_to.

By default it delegates to the user assigner_cp.
